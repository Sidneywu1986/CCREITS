// ==================== 对比栏管理 ====================

function getCompareList() {
    const stored = sessionStorage.getItem('compareList');
    return stored ? JSON.parse(stored) : [];
}

function saveCompareList(list) {
    sessionStorage.setItem('compareList', JSON.stringify(list));
    updateCompareBadge();
    updateCompareBar();
}

function addToCompare(code, name, event) {
    if (event) event.stopPropagation();
    const list = getCompareList();
    
    if (list.find(item => item.code === code)) {
        removeFromCompare(code);
        return;
    }
    
    if (list.length >= 4) {
        showToast('对比栏已满（最多4只）', 'warning');
        return;
    }
    
    list.push({ code, name });
    saveCompareList(list);
    showToast(`已添加 ${name}`, 'success');
}

function removeFromCompare(code) {
    let list = getCompareList();
    list = list.filter(item => item.code !== code);
    saveCompareList(list);
    showToast('已移除', 'info');
}

function clearCompare() {
    saveCompareList([]);
    showToast('已清空对比栏', 'info');
}

function updateCompareBadge() {
    const list = getCompareList();
    const badge = document.getElementById('compare-badge');
    if (badge) {
        if (list.length > 0) {
            badge.textContent = list.length;
            badge.classList.remove('hidden');
        } else {
            badge.classList.add('hidden');
        }
    }
}

function updateCompareBar() {
    const list = getCompareList();
    const bar = document.getElementById('compare-bar');
    const items = document.getElementById('compare-items');
    const countSpan = document.getElementById('compare-count');
    const compareBtn = document.getElementById('btn-compare');
    
    if (!bar) return;
    
    if (list.length === 0) {
        bar.classList.remove('show');
        return;
    }
    
    bar.classList.add('show');
    
    if (items) {
        items.innerHTML = list.map(item => `
            <div class="flex items-center gap-2 px-3 py-1.5 bg-blue-50 border border-blue-200 rounded-lg flex-none">
                <span class="text-sm text-blue-900 font-medium truncate max-w-[120px]">${item.name}</span>
                <button onclick="removeFromCompare('${item.code}')" class="text-blue-400 hover:text-blue-600">
                    <svg class="w-4 h-4" fill="none" stroke="currentColor" viewBox="0 0 24 24"><path stroke-linecap="round" stroke-linejoin="round" stroke-width="2" d="M6 18L18 6M6 6l12 12"/></svg>
                </button>
            </div>
        `).join('');
    }
    
    if (countSpan) countSpan.textContent = `已选 ${list.length}/4`;
    if (compareBtn) {
        compareBtn.disabled = list.length < 2;
        compareBtn.classList.toggle('opacity-50', list.length < 2);
        compareBtn.classList.toggle('cursor-not-allowed', list.length < 2);
    }
}

function goToCompare() {
    const list = getCompareList();
    if (list.length < 2) {
        showToast('请至少选择2只基金进行对比', 'warning');
        return;
    }
    window.location.href = './compare.html';
}

// ==================== Toast提示 ====================

function showToast(message, type = 'info') {
    const existingToast = document.querySelector('.toast');
    if (existingToast) existingToast.remove();
    
    const toast = document.createElement('div');
    toast.className = `toast toast-${type}`;
    toast.textContent = message;
    document.body.appendChild(toast);
    
    setTimeout(() => toast.classList.add('show'), 10);
    setTimeout(() => {
        toast.classList.remove('show');
        setTimeout(() => toast.remove(), 300);
    }, 2500);
}

// ==================== 格式化函数 ====================

function formatNumber(num, decimals = 2) {
    if (num === undefined || num === null) return '--';
    return Number(num).toLocaleString('zh-CN', {
        minimumFractionDigits: decimals,
        maximumFractionDigits: decimals
    });
}

function formatPercent(num, decimals = 2) {
    if (num === undefined || num === null) return '--';
    const sign = num > 0 ? '+' : '';
    return `${sign}${num.toFixed(decimals)}%`;
}

function formatDate(date) {
    if (!date) return '--';
    return date;
}

function getChangeClass(change) {
    if (change > 0) return 'text-up';
    if (change < 0) return 'text-down';
    return '';
}

// ==================== 侧边栏控制 ====================

function toggleSidebar() {
    const sidebar = document.querySelector('.sidebar');
    if (sidebar) sidebar.classList.toggle('open');
}

// ==================== 防抖函数 ====================

function debounce(func, wait) {
    let timeout;
    return function(...args) {
        clearTimeout(timeout);
        timeout = setTimeout(() => func.apply(this, args), wait);
    };
}

    <script>
        // 自选股存储
        let watchlist = [];
        let batchMode = false;
        let selectedItems = new Set();
        
        // 加载自选股
        function loadWatchlist() {
            const stored = localStorage.getItem('watchlist');
            if (stored) {
                watchlist = JSON.parse(stored);
            } else {
                // 默认添加几只示例
                watchlist = [
                    { code: '508056', name: '中金普洛斯REIT', addTime: new Date().toISOString() },
                    { code: '180201', name: '平安广州广河REIT', addTime: new Date().toISOString() },
                    { code: '508099', name: '建信中关村REIT', addTime: new Date().toISOString() }
                ];
                saveWatchlist();
            }
            renderWatchlist();
        }
        
        // 保存自选股
        function saveWatchlist() {
            localStorage.setItem('watchlist', JSON.stringify(watchlist));
            updateStats();
        }
        
        // 获取自选基金的实时数据
        function getWatchlistFunds() {
            return watchlist.map(item => {
                const fund = ALL_FUNDS.find(f => f.code === item.code);
                return {
                    ...item,
                    ...fund,
                    price: fund ? fund.price : 0,
                    change: fund ? fund.change : 0,
                    sector: fund ? fund.sector : '',
                    sectorName: fund ? fund.sectorName : '',
                    yield: fund ? fund.yield : 0,
                    marketCap: fund ? fund.marketCap : 0
                };
            }).filter(f => f.code);
        }
        
        // 更新统计
        function updateStats() {
            const funds = getWatchlistFunds();
            const count = funds.length;
            const totalMarketCap = funds.reduce((sum, f) => sum + (f.marketCap || 0), 0);
            const avgYield = funds.reduce((sum, f) => sum + (f.yield || 0), 0) / (count || 1);
            const totalChange = funds.reduce((sum, f) => sum + (f.change || 0), 0);
            
            document.getElementById('stat-count').textContent = count;
            document.getElementById('stat-market-cap').textContent = totalMarketCap.toFixed(0) + '亿';
            document.getElementById('stat-avg-yield').textContent = avgYield.toFixed(1) + '%';
            document.getElementById('stat-total-change').innerHTML = totalChange > 0 ? 
                `<span class="text-red-600">+${totalChange.toFixed(2)}%</span>` : 
                `<span class="text-green-600">${totalChange.toFixed(2)}%</span>`;
            document.getElementById('watchlist-count').textContent = count + '只自选';
            document.getElementById('list-count').textContent = count;
        }
        
        // 渲染自选列表
        function renderWatchlist() {
            const funds = getWatchlistFunds();
            const compareList = getCompareList();
            const tbody = document.getElementById('watchlist-body');
            const checkboxHeader = document.getElementById('checkbox-header');
            const selectAllCheckbox = document.getElementById('select-all');
            
            if (funds.length === 0) {
                tbody.innerHTML = `
                    <tr>
                        <td colspan="9" class="text-center py-12">
                            <div class="empty-state">
                                <div class="text-5xl mb-3">⭐</div>
                                <div class="text-gray-500 mb-2">暂无自选基金</div>
                                <div class="text-sm text-gray-400">去市场页面点击 ⭐ 添加自选</div>
                                <a href="./market.html" class="inline-block mt-4 btn-primary">去市场添加</a>
                            </div>
                        </td>
                    </tr>
                `;
                checkboxHeader.style.display = 'none';
                return;
            }
            
            checkboxHeader.style.display = batchMode ? 'table-cell' : 'none';
            
            tbody.innerHTML = funds.map((fund, index) => {
                const inCompare = compareList.find(c => c.code === fund.code);
                const sectorClass = SECTOR_CONFIG[fund.sector]?.tagClass || 'sector-tag';
                const isSelected = selectedItems.has(fund.code);
                
                return `
                    <tr class="portfolio-item ${isSelected ? 'bg-blue-50' : ''}" draggable="true" data-index="${index}" data-code="${fund.code}">
                        ${batchMode ? `
                            <td class="w-10">
                                <input type="checkbox" class="select-item" data-code="${fund.code}" ${isSelected ? 'checked' : ''}>
                            </td>
                        ` : ''}
                        <td class="w-10 drag-handle text-gray-400">
                            <svg class="w-4 h-4" fill="none" stroke="currentColor" viewBox="0 0 24 24"><path stroke-linecap="round" stroke-linejoin="round" stroke-width="2" d="M4 8h16M4 16h16"/></svg>
                        </td>
                        <td class="font-mono text-blue-600 font-medium">${fund.code}</td>
                        <td class="font-medium">${fund.name}</td>
                        <td><span class="sector-tag ${sectorClass}">${fund.sectorName}</span></td>
                        <td class="font-mono font-medium" id="price-${fund.code}">${fund.price.toFixed(3)}</td>
                        <td class="font-mono ${getChangeClass(fund.change)}" id="change-${fund.code}">${formatPercent(fund.change)}</td>
                        <td class="font-mono text-red-600">${fund.yield}%</td>
                        <td>
                            <div class="flex items-center gap-1">
                                <button onclick="addToCompare('${fund.code}', '${fund.name}', event)" 
                                    class="p-1.5 rounded ${inCompare ? 'text-blue-600 bg-blue-50' : 'text-gray-400 hover:text-blue-600'}">
                                    <svg class="w-4 h-4" fill="none" stroke="currentColor" viewBox="0 0 24 24"><path stroke-linecap="round" stroke-linejoin="round" stroke-width="2" d="M9 12l2 2 4-4m6 2a9 9 0 11-18 0 9 9 0 0118 0z"/></svg>
                                </button>
                                <button onclick="removeFromWatchlist('${fund.code}')" class="p-1.5 text-gray-400 hover:text-red-500">
                                    <svg class="w-4 h-4" fill="none" stroke="currentColor" viewBox="0 0 24 24"><path stroke-linecap="round" stroke-linejoin="round" stroke-width="2" d="M19 7l-.867 12.142A2 2 0 0116.138 21H7.862a2 2 0 01-1.995-1.858L5 7m5 4v6m4-6v6m1-10V4a1 1 0 00-1-1h-4a1 1 0 00-1 1v3M4 7h16"/></svg>
                                </button>
                            </div>
                        </td>
                    </tr>
                `;
            }).join('');
            
            if (batchMode && selectAllCheckbox) {
                selectAllCheckbox.checked = selectedItems.size === funds.length;
            }
            
            attachDragEvents();
        }
        
        // 移除自选
        function removeFromWatchlist(code) {
            watchlist = watchlist.filter(item => item.code !== code);
            selectedItems.delete(code);
            saveWatchlist();
            renderWatchlist();
            showToast('已移除自选', 'info');
        }
        
        // 批量删除
        function batchDelete() {
            if (selectedItems.size === 0) {
                showToast('请先选择要删除的基金', 'warning');
                return;
            }
            
            if (confirm(`确定删除 ${selectedItems.size} 只自选基金吗？`)) {
                watchlist = watchlist.filter(item => !selectedItems.has(item.code));
                selectedItems.clear();
                saveWatchlist();
                renderWatchlist();
                showToast('删除成功', 'success');
            }
        }
        
        // 导出CSV
        function exportCSV() {
            const funds = getWatchlistFunds();
            if (funds.length === 0) {
                showToast('暂无自选数据可导出', 'warning');
                return;
            }
            
            const headers = ['代码', '名称', '板块', '最新价', '涨跌幅', '派息率', '债务率', '市值'];
            const rows = funds.map(f => [
                f.code, f.name, f.sectorName, f.price, f.change + '%', f.yield + '%', f.debt + '%', f.marketCap + '亿'
            ]);
            
            const csvContent = [headers, ...rows].map(row => row.join(',')).join('\n');
            const blob = new Blob(['\uFEFF' + csvContent], { type: 'text/csv;charset=utf-8;' });
            const link = document.createElement('a');
            const url = URL.createObjectURL(blob);
            link.href = url;
            link.setAttribute('download', `reits_watchlist_${new Date().toISOString().slice(0,10)}.csv`);
            document.body.appendChild(link);
            link.click();
            document.body.removeChild(link);
            URL.revokeObjectURL(url);
            showToast('导出成功', 'success');
        }
        
        // 刷新行情
        function refreshQuotes() {
            // 模拟价格波动
            watchlist.forEach(item => {
                const fund = ALL_FUNDS.find(f => f.code === item.code);
                if (fund) {
                    const change = (Math.random() - 0.5) * 0.3;
                    fund.price = parseFloat((fund.price * (1 + change / 100)).toFixed(3));
                    fund.change = parseFloat((fund.change + change).toFixed(2));
                    
                    // 更新DOM
                    const priceEl = document.getElementById(`price-${fund.code}`);
                    const changeEl = document.getElementById(`change-${fund.code}`);
                    if (priceEl) {
                        priceEl.textContent = fund.price.toFixed(3);
                        priceEl.classList.add('flash-up');
                        setTimeout(() => priceEl.classList.remove('flash-up'), 500);
                    }
                    if (changeEl) {
                        changeEl.textContent = formatPercent(fund.change);
                        changeEl.className = `font-mono ${getChangeClass(fund.change)}`;
                    }
                }
            });
            updateStats();
            showToast('行情已刷新', 'success');
        }
        
        // 拖拽排序
        function attachDragEvents() {
            const items = document.querySelectorAll('#watchlist-body tr[draggable="true"]');
            let dragSrc = null;
            
            items.forEach(item => {
                item.addEventListener('dragstart', (e) => {
                    dragSrc = item;
                    e.dataTransfer.effectAllowed = 'move';
                    item.style.opacity = '0.5';
                });
                
                item.addEventListener('dragend', (e) => {
                    item.style.opacity = '';
                    items.forEach(i => i.classList.remove('drag-over'));
                });
                
                item.addEventListener('dragover', (e) => {
                    e.preventDefault();
                    e.dataTransfer.dropEffect = 'move';
                });
                
                item.addEventListener('dragenter', (e) => {
                    e.preventDefault();
                    if (dragSrc !== item) {
                        item.classList.add('drag-over');
                    }
                });
                
                item.addEventListener('dragleave', (e) => {
                    item.classList.remove('drag-over');
                });
                
                item.addEventListener('drop', (e) => {
                    e.preventDefault();
                    if (dragSrc !== item) {
                        const srcIndex = parseInt(dragSrc.dataset.index);
                        const targetIndex = parseInt(item.dataset.index);
                        
                        const newWatchlist = [...watchlist];
                        const [moved] = newWatchlist.splice(srcIndex, 1);
                        newWatchlist.splice(targetIndex, 0, moved);
                        watchlist = newWatchlist;
                        saveWatchlist();
                        renderWatchlist();
                        showToast('排序已保存', 'success');
                    }
                    item.classList.remove('drag-over');
                    dragSrc.style.opacity = '';
                });
            });
        }
        
        // 批量选择模式
        function toggleBatchMode() {
            batchMode = !batchMode;
            if (!batchMode) {
                selectedItems.clear();
            }
            const batchDeleteBtn = document.getElementById('batch-delete-btn');
            const cancelSelectBtn = document.getElementById('cancel-select-btn');
            const batchSelectBtn = document.getElementById('batch-select-btn');
            
            if (batchMode) {
                batchDeleteBtn.classList.remove('hidden');
                cancelSelectBtn.classList.remove('hidden');
                batchSelectBtn.classList.add('hidden');
            } else {
                batchDeleteBtn.classList.add('hidden');
                cancelSelectBtn.classList.add('hidden');
                batchSelectBtn.classList.remove('hidden');
            }
            
            renderWatchlist();
        }
        
        // 全选/取消全选
        function toggleSelectAll() {
            const funds = getWatchlistFunds();
            const selectAll = document.getElementById('select-all');
            if (selectAll.checked) {
                funds.forEach(f => selectedItems.add(f.code));
            } else {
                selectedItems.clear();
            }
            document.getElementById('selected-count').textContent = selectedItems.size;
            renderWatchlist();
        }
        
        // 全局搜索
        window.handleGlobalSearch = (keyword) => {
            const funds = getWatchlistFunds();
            const filtered = funds.filter(f => 
                f.code.toLowerCase().includes(keyword) || 
                f.name.toLowerCase().includes(keyword)
            );
            // 简单实现：重新渲染过滤后的列表
            const tbody = document.getElementById('watchlist-body');
            if (keyword && filtered.length < funds.length) {
                // 显示过滤结果
                tbody.innerHTML = filtered.map(fund => {
                    const sectorClass = SECTOR_CONFIG[fund.sector]?.tagClass || 'sector-tag';
                    return `
                        <tr>
                            <td class="w-10"></td>
                            <td class="font-mono text-blue-600">${fund.code}</td>
                            <td>${fund.name}</td>
                            <td><span class="sector-tag ${sectorClass}">${fund.sectorName}</span></td>
                            <td>${fund.price.toFixed(3)}</td>
                            <td class="${getChangeClass(fund.change)}">${formatPercent(fund.change)}</td>
                            <td>${fund.yield}%</td>
                            <td>
                                <button onclick="removeFromWatchlist('${fund.code}')" class="text-gray-400 hover:text-red-500">删除</button>
                            </td>
                        </tr>
                    `;
                }).join('');
            } else {
                renderWatchlist();
            }
        };
        
        // 事件绑定
        document.getElementById('batch-select-btn')?.addEventListener('click', toggleBatchMode);
        document.getElementById('batch-delete-btn')?.addEventListener('click', batchDelete);
        document.getElementById('cancel-select-btn')?.addEventListener('click', toggleBatchMode);

        function calculateDimensions(fund) {
            // 流动性：基于成交量（0-100）
            const liquidity = Math.min(100, Math.max(0, (fund.volume / 50000) * 100));
            // 稳定性：基于涨跌幅波动（越小越稳定）
            const stability = Math.min(100, Math.max(0, 100 - Math.abs(fund.change) * 5));
            // 成长性：基于派息率和溢价率
            const growth = Math.min(100, Math.max(0, (fund.yield - 3) * 20 + (fund.premium > 0 ? 10 : -10)));
            // 安全性：基于债务率（越低越安全）
            const safety = Math.min(100, Math.max(0, 100 - fund.debt));
            // 收益性：基于派息率
            const profitability = Math.min(100, Math.max(0, (fund.yield - 3) * 25));
            
            return { liquidity, stability, growth, safety, profitability };
        }

        function renderCompareTable() {
            if (compareList.length === 0) return;

            // 渲染表头
            const thead = document.getElementById('table-header');

            const metrics = [
                { label: '📊 最新价', key: 'price', format: (v) => `${v.toFixed(3)}元`, higherBetter: false },
                { label: '📈 当日涨跌', key: 'change', format: (v) => formatPercent(v), higherBetter: true },
                { label: '💰 派息率(TTM)', key: 'yield', format: (v) => `${v}%`, higherBetter: true },
                { label: '🏷️ 溢价率', key: 'premium', format: (v) => `${v > 0 ? '+' : ''}${v}%`, higherBetter: false },
                { label: '📉 资产负债率', key: 'debt', format: (v) => `${v}%`, higherBetter: false },
                { label: '📦 成交量', key: 'volume', format: (v) => `${(v / 10000).toFixed(0)}万手`, higherBetter: true },
                { label: '🏢 管理规模', key: 'scale', format: (v) => `${v}亿`, higherBetter: true },
                { label: '💎 流通市值', key: 'marketCap', format: (v) => `${v}亿`, higherBetter: true },
                { label: '📅 剩余期限', key: 'remainingYears', format: (v) => v, higherBetter: false },
                { label: '🏗️ 资产类型', key: 'propertyType', format: (v) => v || '--', higherBetter: false }
            ];
            
            // 找出每项指标的最优/最差值
            const bestValues = {};
            const worstValues = {};
            metrics.forEach(metric => {
                const values = funds.map(f => f[metric.key]).filter(v => typeof v === 'number');
                if (values.length > 0) {
                    bestValues[metric.key] = metric.higherBetter ? Math.max(...values) : Math.min(...values);
                    worstValues[metric.key] = metric.higherBetter ? Math.min(...values) : Math.max(...values);
                }
            });
            
            // 渲染表格行
            const tbody = document.getElementById('table-body');
            tbody.innerHTML = metrics.map((metric, idx) => {
                const cells = funds.map(fund => {
                    let value = fund[metric.key];
                    let formatted = metric.format(value);
                    let additionalClass = '';
                    let highlightHtml = '';
                    
                    if (highlightBest && typeof value === 'number') {
                        if (value === bestValues[metric.key]) {
                            additionalClass = 'highlight-best';
                            highlightHtml = '<span class="text-xs ml-1">🏆</span>';
                        } else if (value === worstValues[metric.key] && metric.key !== 'remainingYears') {
                            additionalClass = 'highlight-worst';
                        }
                    }
                    
                    return `<td class="px-4 py-3 text-center ${additionalClass}">${formatted}${highlightHtml}</td>`;
                }).join('');
                
                return `
                    <tr class="compare-row">
                        <td class="px-4 py-3 font-medium text-gray-700 bg-gray-50">${metric.label}</td>
                        ${cells}
                    </tr>
                `;
            }).join('');

        function initFundSelectors() {

        // 收益率计算
        function calculateYield() {
            const buyPrice = parseFloat(document.getElementById('buy-price').value) || 0;
            const sellPrice = parseFloat(document.getElementById('sell-price').value) || 0;
            const dividend = parseFloat(document.getElementById('dividend-received').value) || 0;
            const years = parseFloat(document.getElementById('hold-years').value) || 1;
            
            const capitalGain = sellPrice - buyPrice;
            const totalGain = capitalGain + dividend;
            const totalReturn = (totalGain / buyPrice) * 100;
            const annualReturn = (Math.pow(1 + totalReturn / 100, 1 / years) - 1) * 100;
            
            document.getElementById('total-return').textContent = totalReturn.toFixed(2) + '%';
            document.getElementById('annual-return').textContent = annualReturn.toFixed(2) + '%';
            document.getElementById('capital-gain').textContent = (capitalGain > 0 ? '+' : '') + capitalGain.toFixed(3) + '元';
            document.getElementById('dividend-gain').textContent = '+' + dividend.toFixed(3) + '元';
        }

        // 生成模拟分红数据
        function generateDividendData() {
            const dividends = [];
            const startDate = new Date('2025-01-01');
            const endDate = new Date('2026-12-31');
            
            for (let i = 0; i < 180; i++) {
                const fund = ALL_FUNDS[Math.floor(Math.random() * ALL_FUNDS.length)];
                const date = new Date(startDate);
                date.setDate(startDate.getDate() + Math.floor(Math.random() * 730));
                
                if (date <= endDate) {
                    dividends.push({
                        id: i,
                        code: fund.code,
                        name: fund.name,
                        exDate: date.toISOString().split('T')[0],
                        recordDate: new Date(date.getTime() + 2 * 86400000).toISOString().split('T')[0],
                        paymentDate: new Date(date.getTime() + 14 * 86400000).toISOString().split('T')[0],
                        amount: parseFloat((0.3 + Math.random() * 1.2).toFixed(2)),
                        yield: fund.yield
                    });
                }
            }
            
            dividends.sort((a, b) => new Date(a.exDate) - new Date(b.exDate));
            return dividends;
        }
        
        const DIVIDEND_DATA = generateDividendData();
        let currentYear = 2026;
        let currentMonth = 3; // 3月
        
        // 获取某天的分红
        function getDividendsForDate(year, month, day) {
            const dateStr = `${year}-${String(month).padStart(2, '0')}-${String(day).padStart(2, '0')}`;
            return DIVIDEND_DATA.filter(d => d.exDate === dateStr);
        }
        
        // 渲染日历
        function renderCalendar() {
            const firstDay = new Date(currentYear, currentMonth - 1, 1);
            const lastDay = new Date(currentYear, currentMonth, 0);
            const startWeekday = firstDay.getDay(); // 0=周日
            const daysInMonth = lastDay.getDate();
            const today = new Date();
            const isCurrentMonth = today.getFullYear() === currentYear && today.getMonth() + 1 === currentMonth;
            
            let gridHtml = '';
            
            // 填充空白
            for (let i = 0; i < startWeekday; i++) {
                gridHtml += `<div class="aspect-square p-2 text-gray-300"></div>`;
            }
            
            // 填充日期
            for (let day = 1; day <= daysInMonth; day++) {
                const dividends = getDividendsForDate(currentYear, currentMonth, day);
                const hasDividend = dividends.length > 0;
                const isToday = isCurrentMonth && today.getDate() === day;
                const dayClass = `aspect-square p-2 rounded-xl calendar-day ${hasDividend ? 'has-dividend' : ''} ${isToday ? 'today' : ''} hover:bg-gray-50`;
                
                let dividendHtml = '';
                if (hasDividend && dividends.length <= 2) {
                    dividendHtml = dividends.map(d => `
                        <div class="text-xs text-orange-600 truncate mt-1" title="${d.name} ${d.amount}元/份">
                            💰 ${d.amount}元
                        </div>
                    `).join('');
                } else if (hasDividend && dividends.length > 2) {
                    dividendHtml = `<div class="text-xs text-orange-600 mt-1">💰 ${dividends.length}只分红</div>`;
                }
                
                gridHtml += `
                    <div class="${dayClass}" onclick="showDividendDetail(${day})">
                        <div class="font-medium ${isToday ? 'text-blue-600' : ''}">${day}</div>
                        ${dividendHtml}
                    </div>
                `;
            }
            
            document.getElementById('calendar-grid').innerHTML = gridHtml;
            document.getElementById('current-month').textContent = `${currentYear}年${currentMonth}月`;
        }

                            <div class="text-red-600 font-bold text-lg">${d.amount}元</div>
                            <div class="countdown text-xs ${daysLeft <= 3 ? 'text-red-500 font-bold' : 'text-gray-400'}">
                                ${daysLeft <= 0 ? '今日除息' : daysLeft === 1 ? '明天除息' : `${daysLeft}天后除息`}
                            </div>
                        </div>
                    </div>
                `;
            }).join('');
        }
        
        // 月度分红统计图表
        function renderMonthlyChart() {
            const monthlyData = {};
            for (let m = 1; m <= 12; m++) {
                monthlyData[m] = 0;
            }
            
            DIVIDEND_DATA.forEach(d => {
                const month = parseInt(d.exDate.split('-')[1]);
                monthlyData[month] += d.amount;
            });
            
            const chart = echarts.init(document.getElementById('monthly-chart'));
            chart.setOption({
                tooltip: { trigger: 'axis' },
                xAxis: { type: 'category', data: ['1月', '2月', '3月', '4月', '5月', '6月', '7月', '8月', '9月', '10月', '11月', '12月'] },
                yAxis: { type: 'value', name: '分红金额 (元/份)' },
                series: [{
                    type: 'bar', data: Object.values(monthlyData),
                    itemStyle: {
                        borderRadius: [8, 8, 0, 0],
                        color: { type: 'linear', x: 0, y: 0, x2: 0, y2: 1,
                            colorStops: [{ offset: 0, color: '#3b82f6' }, { offset: 1, color: '#8b5cf6' }]
                        }
                    }
                }]
            });
        }
        
        // 高分红排行
        function renderRanking() {
            const fundDividendMap = {};
            DIVIDEND_DATA.forEach(d => {
                if (!fundDividendMap[d.code]) {
                    fundDividendMap[d.code] = { name: d.name, total: 0, count: 0 };
                }
                fundDividendMap[d.code].total += d.amount;
                fundDividendMap[d.code].count++;
            });
            
            const ranking = Object.entries(fundDividendMap)
                .map(([code, data]) => ({ code, ...data, avg: data.total / data.count }))
                .sort((a, b) => b.total - a.total)
                .slice(0, 10);
            
            const container = document.getElementById('ranking-list');
            container.innerHTML = ranking.map((item, idx) => `
                <div class="flex items-center justify-between p-3 rounded-xl hover:bg-gray-50 cursor-pointer" onclick="window.location.href='./fund-archive.html?code=${item.code}'">
                    <div class="flex items-center gap-3">
                        <div class="w-8 h-8 rounded-full ${idx === 0 ? 'bg-yellow-100 text-yellow-600' : idx === 1 ? 'bg-gray-100 text-gray-600' : idx === 2 ? 'bg-orange-100 text-orange-600' : 'bg-gray-50 text-gray-400'} flex items-center justify-center font-bold">
                            ${idx + 1}
                        </div>
                        <div>
                            <div class="font-medium">${item.name}</div>
                            <div class="text-xs text-gray-400 font-mono">${item.code}</div>
                        </div>
                    </div>
                    <div class="text-right">
                        <div class="font-bold text-red-600">${item.total.toFixed(2)}元</div>
                        <div class="text-xs text-gray-400">年均 ${item.avg.toFixed(2)}元</div>
                    </div>
                </div>
            `).join('');
        }
        
        // 月份切换
        function changeMonth(delta) {
            let newMonth = currentMonth + delta;
            let newYear = currentYear;
            
            if (newMonth < 1) {
                newMonth = 12;
                newYear--;
            } else if (newMonth > 12) {
                newMonth = 1;
                newYear++;
            }
            
            currentYear = newYear;
            currentMonth = newMonth;
            renderCalendar();
        }
        
        function goToToday() {
            const today = new Date();
            currentYear = today.getFullYear();
            currentMonth = today.getMonth() + 1;
            renderCalendar();
        }
        
        // 事件绑定
        document.getElementById('prev-month')?.addEventListener('click', () => changeMonth(-1));
        document.getElementById('next-month')?.addEventListener('click', () => changeMonth(1));
        document.getElementById('today-btn')?.addEventListener('click', goToToday);
        
        // 初始化
        renderCalendar();
        renderUpcomingDividends();
        renderMonthlyChart();
        renderRanking();
    </script>
</body>
